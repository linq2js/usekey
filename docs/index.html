objkey

Object key generator

Installation

with NPM

npm i objkey --save

with YARN

yarn add objkey

Usages

 
import keyOf from "objkey";

// a todo list without key/id prop
const todos = [{ title: "Todo1" }, { title: "Todo2" }, { title: "Todo3" }];

export default function App() {
  return (
    <div className="App">
      {todos.map((todo) => (
        // generate key for todo object
        <div key={keyOf(todo)}>
          {keyOf(todo)}:{todo.title}
        </div>
      ))}
    </div>
  );
}
